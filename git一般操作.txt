git add .
git status
git commit -m '代码说明'
git show 
#提交代码远程
git push origin master
#远程拉取代码
git pull origin master
#分部拉取代码
git fetch
git diff origin/master
git merge origin

#冲突
git stash
git pull   origin master
git stash pop stash@{0}
git checkout ./   --更新目录
git stash pop

#创建分支
git checkout -b wms
#推送绑定分支
git push origin wms:wms
#绑定远程分支
git branch --set-upstream-to=origin/wms wms


删除分支
git git branch -d wms
删除远程分支
git push origin :wms


#合并
git merge --no-ff  wms 

#线性合并
git rebase master
git add .
git rebase --continue 继续
git rebase --abort    取消

#合并日志节点
git cherry-pick 99daed2

#取消合并
git reset --hard HEAD~

#取消提交
git reset --soft 
git reset --mixed 	
git reset --hard  #删除数据


分支思想
master 线上主分支
	hotfix-* 紧急BUG修复分支 ->线上,开发 
	release-* 发布分支 ->线上,开发 git merge --no-ff release-1.2  git tag -a 1.2
develop 开发主分支
	feature 功能分支 ->开发 git merge --no-ff myfeature
