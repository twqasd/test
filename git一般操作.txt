git add .
git status
git commit -m '代码说明'
git show 
#提交代码远程
git push origin master
#远程拉取代码
git pull origin master
#分部拉取代码
git fetch
git diff origin/master
git merge origin

#冲突
git stash
git pull   origin master
git stash pop stash@{0}

#创建分支
git checkout -b wms
#推送绑定分支
git push origin wms:wms

删除分支
git git branch -d wms
删除远程分支
git push origin :wms


#合并
git merge --no-ff  wms 

#线性合并
git rebase wms 

#合并日志节点
git cherry-pick 99daed2
